const express = require('express');
const app = express();
const cookieParser = require('cookie-parser');

app.use(cookieParser());

app.get("/", function (req, res) {
    res.cookie("name", "harshita");
    res.send("done");
})

app.get("/read", function (req, res) {
    console.log(req.cookies);
    res.send("read page ");
})

app.listen(3000);
